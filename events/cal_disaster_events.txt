country_event = {#start of the disaster
	id = cal_disaster.1
	title = cal_disaster.1.t
	desc = cal_disaster.1.d
	picture = REVOLUTION_eventPicture

	is_triggered_only = yes

	major = yes

	immediate = {
		set_country_flag = cal_has_illegimitate_caliph
		random_owned_province = {
			limit = {
				NOT = { has_province_flag = cal_disaster_spawn_rebels_here }
				NOT = { units_in_province = 1 }
				is_capital = no
			}
			set_province_flag = cal_disaster_spawn_rebels_here
		}
		if = {
			limit = {
				num_of_cities = 30
			}
			random_owned_province = {
				limit = {
					NOT = { has_province_flag = cal_disaster_spawn_rebels_here }
					NOT = { units_in_province = 1 }
					is_capital = no
				}
				set_province_flag = cal_disaster_spawn_rebels_here
			}
		}

		restore_country_name = yes
	}

	option = { #caliphate
		name = cal_disaster.1.a

		set_country_flag = cal_caliphate_fight

		random_owned_province = {
			limit = {
				has_province_flag = cal_disaster_spawn_rebels_here
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 3
			}
			clr_province_flag = cal_disaster_spawn_rebels_here
		}
		if = {
			limit = {
				any_owned_province = {
					has_province_flag = cal_disaster_spawn_rebels_here
				}
			}
			random_owned_province = {
				limit = {
					has_province_flag = cal_disaster_spawn_rebels_here
				}
				spawn_rebels = {
					type = pretender_rebels
					size = 2
				}
				clr_province_flag = cal_disaster_spawn_rebels_here
			}
		}
	}

	option = { #abandon caliphate, become atheist
		name = cal_disaster.1.b

		set_country_flag = cal_atheist_state

		random_owned_province = {
			limit = {
				has_province_flag = cal_disaster_spawn_rebels_here
			}
			spawn_rebels = {
				type = pretender_rebels
				use_monarch_as_leader = yes
				size = 3
			}
			clr_province_flag = cal_disaster_spawn_rebels_here
		}
		if = {
			limit = {
				any_owned_province = {
					has_province_flag = cal_disaster_spawn_rebels_here
				}
			}
			random_owned_province = {
				limit = {
					has_province_flag = cal_disaster_spawn_rebels_here
				}
				spawn_rebels = {
					type = pretender_rebels
					size = 2
				}
				clr_province_flag = cal_disaster_spawn_rebels_here
			}
		}

		if = {
			limit = {
				NOT = { government = republic }
			}
			change_government_to_republic = yes
		}
		add_government_reform = military_dictatorship_reform

		kill_ruler = yes
	}
}

		}

			limit = {
				development_in_provinces = {
					value = 500

				}
			}
			add_years_of_income = 5
			add_yearly_manpower = 5
		}
		else_if = {
			limit = {
				development_in_provinces = {
					value = 300
					owned_by = ROOT
					culture = event_target:genocide_province
				}
			}
			add_years_of_income = 3
			add_yearly_manpower = 3
		}

			limit = {
				development_in_provinces = {
					value = 100
					owned_by = ROOT
					culture = event_target:genocide_province
				}
			}
			add_years_of_income = 2
			add_yearly_manpower = 2

			add_years_of_income = 1
			add_yearly_manpower = 1
		}

		add_war_exhaustion = -3
		custom_tooltip = cal_warning_genocide
		every_owned_province = {
			limit = {
				culture = event_target:genocide_province
			}
			add_province_modifier = {
				name = cal_genocide_province
				duration = 7300
			}
			random_list = {
				1 = {
					add_base_manpower = -3
				}
				1 = {
					add_base_manpower = -2
				}
				1 = {
					add_base_manpower = -1
				}
			}
			random_list = {
				1 = {
					add_base_tax = -3
				}
				1 = {
					add_base_tax = -2
				}
				1 = {
					add_base_tax = -1
				}
			}
			random_list = {
				1 = {
					add_base_production = -3
				}
				1 = {
					add_base_production = -2
				}
				1 = {
					add_base_production = -1
				}
			}
		}
		hidden_effect = { # spawn repurcussion event in 30 - 50 years
			random_list = {
				1 = {
					country_event = {
						id = cal_flavor.7
						days = 10800
						random = 7300
					}
					modifier = {
						factor = 2
						num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
							value = 10
							owned_by = ROOT
							culture = event_target:genocide_province
						}
					}
					modifier = {
						factor = 2
						num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
							value = 20
							owned_by = ROOT
							culture = event_target:genocide_province
						}
					}
					modifier = {
						factor = 2
						development_in_provinces = {
							value = 100
							owned_by = ROOT
							culture = event_target:genocide_province
						}
					}
					modifier = {
						factor = 3
						development_in_provinces = {
							value = 300
							owned_by = ROOT
							culture = event_target:genocide_province
						}
					}
				}
				1 = {
					country_event = {
						id = cal_flavor.8
						days = 10800
						random = 7300
					}
				}
			}
		}
	}

	option = { # leave them be
		name = cal_disaster.2.b
		add_stability = -1
		add_war_exhaustion = 5
	}
