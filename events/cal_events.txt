country_event = {#unify islam
	id = cal_flavor.5
	title = cal_flavor.5.t
	desc = cal_flavor.5.d
	picture = HAJJ_eventPicture

	is_triggered_only = yes

	major = yes

	option = {#islamic state
		name = cal_flavor.5.a
		change_tag = ISS
		if = {
			limit = { has_custom_ideas = no }
			country_event = { id = ideagroups.1 } #Swap Ideas
		}
		remove_non_electors_emperors_from_empire_effect = yes

		if = {
			limit = {
				NOT = { government = monarchy }
			}
			change_government_to_monarchy = yes
		}
		add_government_reform = cal_religous_state_reform

		add_country_modifier = {
			name = cal_islamic_state
			duration = -1
		}
		every_country = {
			limit = {
				NOT = { religion_group = muslim }
			}
			reverse_add_opinion = {
				who = THIS
				modifier = cal_islamic_state_opinion
			}
		}
	}

	option = {#moderate caliphate
		name = cal_flavor.5.b
		if = {
			limit = {
				NOT = { government = monarchy }
			}
			change_government_to_monarchy = yes
		}
		add_government_reform = cal_caliphate_reform

		add_country_modifier = {
			name = cal_moderate_caliphate
			duration = -1
		}
	}
	option = {#protector
		name = cal_flavor.5.c

		add_country_modifier = {
			name = cal_protector_caliphate
			duration = -1
		}
		every_country = {
			limit = {
				religion_group = muslim
			}
			reverse_add_opinion = {
				who = THIS
				modifier = cal_protector_opinion
			}
		}
	}
}

country_event = {#feminism
	id = cal_flavor.6
	title = cal_flavor.6.t
	desc = cal_flavor.6.d
	picture = WITCHTRIAL_eventPicture

	trigger = {
		has_reform = cal_atheist_republic_reform
		NOT = {
			OR = {
				has_country_modifier = cal_feminism
				has_country_modifier = cal_patriarchy
			}
		}
		stability = 0
	}

	mean_time_to_happen = {
		months = 500
	
		modifier = {
			factor = 0.8
			stability = 1
		}
		
		modifier = {
			factor = 0.8
			stability = 2
		}
		
		modifier = {
			factor = 0.8
			stability = 3
		}

		modifier = {
			factor = 0.1
			is_female = yes
		}
	}
	

	major = yes

	option = {#feminism
		name = cal_flavor.6.a
		add_country_modifier = {
			name = cal_feminism
			duration = -1
		}
		add_stability = -3
		add_adm_power = -100
	}

	option = {#patriarchy
		name = cal_flavor.6.b
		add_country_modifier = {
			name = cal_patriarchy
			duration = -1
		}
		add_stability = 1
	}
}

	}
	option = { #genocide them again to show dominance
		name = cal_flavor.7.c

		trigger = {
			ruler_has_personality = cruel_personality
			NOT = { primary_culture = event_target:genocide_province }
		}
		every_owned_province = {
			limit = {
				culture = event_target:genocide_province
			}
			add_province_modifier = {
				name = cal_genocide_province
				duration = 7300
			}
			add_base_manpower = -5
			add_base_tax = -5

		}

	}
}
